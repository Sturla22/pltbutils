339  function tcfilter(
340  constant s : string
341  ) return string is
342  variable r : string(s'range) := (others => (' '));
343  begin
344  for i in s'range loop
345 
346  if (s(i) = ''') then
347  r(i) := '`';
348  elsif (s(i) = lf or s(i) = cr) then
349  r(i) := ' ';
350  elsif (s(i) = '|') then
351  r(i) := '/';
352  elsif (s(i) = '[') then
353  r(i) := '{';
354  elsif (s(i) = ']') then
355  r(i) := '}';
356  else
357  r(i) := s(i);
358  end if;
359 
360  end loop;
361  return r;
362  end function tcfilter;
