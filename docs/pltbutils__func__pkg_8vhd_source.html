1333  exit when endfile(f);
1334  readline(f, l);
1335  line_num := line_num + 1;
1336  elsif l'length = 0 then
1337  exit when endfile(f);
1338  readline(f, l);
1339  line_num := line_num + 1;
1340  end if;

1343  read(l, c, c_good);
1344  -- Skip leading whitespace characters
1345  while c_good and (c = ' ' or c = ht or c = cr or c = lf) loop
1346  read(l, c, c_good);
1347  end loop;
1348  -- Read item until next whitespace, comment character or end-of-line
1349  while c_good and c /= ' ' and c /= ht and c /= cr and c /= lf and c /= '#' loop
1350  -- If i >= item'length, an error will occur. We don't try to do anything
1351  -- about that here, because it is probably better that the user
1352  -- find out the hard way, instead of trying to hide the problem.
1353  item(item'low + i) := c;
1354  i := i + 1;
1355  read(l, c, c_good);
1356  end loop;

1358  -- If a comment character was found, absorb the rest of the comment
1359  if c_good and c = '#' then
1360  while l'length > 0 loop
1361  read(l, c, c_good);
1362  end loop;
1363  end if;
1364  -- Update output values if item was found
1365  if i > 0 then
1366  item_len := i;
1367  item_num := item_num + 1;
1368  end if;
1369  end loop;
